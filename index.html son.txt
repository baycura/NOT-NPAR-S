var _rulesCache = {};
var _currentEvtId = null; // Şu an seçili etkinlik ID'si
var _currentRulesOriginal = null; // Orijinal Türkçe metin

async function translateRules(text, targetLang, evtId) {
  if (!text || targetLang === 'tr') return text; // Türkçe orijinal, çevirme

  var cacheKey = evtId + '_' + targetLang;
  if (_rulesCache[cacheKey]) return _rulesCache[cacheKey]; // Önbellekten dön

  // Yükleniyor göster
  var rt = document.getElementById('rulesText');
  if (rt) {
    rt.style.opacity = '0.4';
    rt.textContent = text; // Önce orijinali göster
  }

  try {
    var langName = targetLang === 'en' ? 'English' : 'Russian';
    var response = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        model: 'claude-sonnet-4-20250514',
        max_tokens: 1000,
        messages: [{
          role: 'user',
          content: 'Translate the following event rules to ' + langName + '. Keep the same tone and formatting. Return ONLY the translated text, no explanations, no quotes.

' + text
        }]
      })
    });
    var data = await response.json();
    var translated = data.content && data.content[0] ? data.content[0].text.trim() : text;
    _rulesCache[cacheKey] = translated;
    if (rt) rt.style.opacity = '1';
    return translated;
  } catch(e) {
    console.log('Çeviri hatası:', e);
    if (rt) rt.style.opacity = '1';
    return text; // Hata olursa orijinali göster
  }
}

async function applyRulesTranslation() {
  if (!_currentEvtId || !_currentRulesOriginal) return;
  var rw = document.getElementById('rulesWrap');
  var rt = document.getElementById('rulesText');
  if (!rw || !rt || rw.style.display === 'none') return;

  var evtIdAtCall = _currentEvtId;

  // Önbellekte yoksa yükleniyor göster
  var cacheKey = _currentEvtId + '_' + lang;
  if (lang !== 'tr' && !_rulesCache[cacheKey]) {
    rt.style.opacity = '0.35';
    // Dil badge'i göster
    var badge = document.getElementById('rulesLangBadge');
    if (!badge) {
      badge = document.createElement('span');
      badge.id = 'rulesLangBadge';
      badge.style.cssText = 'font-family:Space Mono,monospace;font-size:7px;color:var(--blue);letter-spacing:1px;margin-left:8px;opacity:.7';
      var rulesFirst = rw.querySelector('.rules-text');
      if (rulesFirst) rulesFirst.parentNode.insertBefore(badge, rulesFirst);
    }
    badge.textContent = '⟳ ÇEVİRİLİYOR...';
  }

  var translated = await translateRules(_currentRulesOriginal, lang, _currentEvtId);

  // Etkinlik değişmediyse güncelle
  if (_currentEvtId === evtIdAtCall) {
    rt.textContent = translated;
    rt.style.opacity = '1';

    if (badge2) badge2.textContent = lang === 'tr' ? '' : (lang === 'en' ? '🇬🇧 EN' : '🇷🇺 RU');
  }
}

var refTimer;
async function checkRef(v) {
  clearTimeout(refTimer);
  var ok = document.getElementById('refOk');
  var er = document.getElementById('refErr');
  if (ok) ok.style.display = 'none';
  if (er) er.style.display = 'none';
  refM = null;
  if (!v || v.length < 10) return;
  refTimer = setTimeout(async function() {
    var res = await _SB.from('members').select('*').eq('member_code', v).eq('status', 'approved').single();
    if (res.data) {
      var used = res.data.referral_used || 0;
      var lim = res.data.referral_limit || 3;
      if (used >= lim) {
        if (er) { er.textContent = 'Bu referans kodunun hakkı dolmuş'; er.style.display = 'block'; }
        refM = null;
      } else {
        refM = res.data;
        if (ok) { ok.textContent = '✓ Referans: ' + res.data.name + ' (' + (lim - used) + ' hak kaldı)'; ok.style.display = 'block'; }
      }
    } else {
      if (er) { er.textContent = t('errRef'); er.style.display = 'block'; }
    }
  }, 400);
}

async function doRegister() {

function renderCarousel() {
  var tr = document.getElementById('cTrack');
  var dt = document.getElementById('cDots');
  tr.innerHTML = '';
  dt.innerHTML = '';
  ci = 0;
  if (!events.length) {
    tr.innerHTML = '<div class="empty" style="width:calc(100vw - 60px)">Etkinlik yok.</div>';
    return;
  }
  events.forEach(function(e, i) {
    var c = document.createElement('div');
    c.className = 'pc' + (selEvt === e.id ? ' sel' : '');
    c.dataset.id = e.id;
    c.onclick = function() { selE(e.id); };
    var inner = '';
    if (e.poster_b64) {
      inner += '<img src="' + e.poster_b64 + '" draggable="false">';
    } else {
      inner += '<div style="width:100%;aspect-ratio:4/5;display:flex;align-items:center;justify-content:center;background:#0d0d0d"><span style="font-family:CV,sans-serif;font-size:24px;color:#2a2a2a">' + e.name + '</span></div>';
    }
    var daysLeft = Math.ceil((new Date(e.date + 'T00:00:00') - new Date()) / 86400000);
    var countdown = daysLeft > 0 ? (daysLeft === 1 ? 'Yarın' : daysLeft + ' gün kaldı') : '';
    inner += '<div class="pc-meta"><div class="pc-name">' + e.name + '</div>';
    inner += '<div class="pc-row">';
    if (e.genre) inner += '<span class="pc-genre">' + e.genre + '</span>';
    inner += '<span class="pc-date">' + fmtDate(e.date) + ' · ' + (e.time || '') + '</span>';
    if (countdown) inner += '<span style="font-family:Space Mono,monospace;font-size:8px;color:var(--gold);letter-spacing:1px;margin-left:auto">' + countdown + '</span>';
    inner += '</div></div>';
    inner += '<div class="pc-ring"></div><div class="pc-chk">✓</div>';
    c.innerHTML = inner;
    tr.appendChild(c);
    var d = document.createElement('div');
    d.className = 'c-dot' + (i === 0 ? ' active' : '');
    d.onclick = function() { ci = i; updC(); };
    dt.appendChild(d);
  });
  updC(false);
  setupSwipe();
}

function updC(anim) {
  if (anim === undefined) anim = true;
  var tr = document.getElementById('cTrack');
  var cs = tr.querySelectorAll('.pc');
  if (!cs.length) return;
  var cw = cs[0].offsetWidth + 10;
  tr.style.transition = anim ? 'transform .42s cubic-bezier(.4,0,.2,1)' : 'none';
  tr.style.transform = 'translateX(-' + (ci * cw) + 'px)';
  document.querySelectorAll('.c-dot').forEach(function(d, i) {
    d.classList.toggle('active', i === ci);
  });
}

function setupSwipe() {
  var vp = document.getElementById('cVP');
  vp.ontouchstart = function(e) { sx = e.touches[0].clientX; drag = true; };
  vp.ontouchmove = function(e) { if (drag) e.preventDefault(); };
  vp.ontouchend = function(e) {
    if (!drag) return;
    drag = false;
    var dx = e.changedTouches[0].clientX - sx;
    if (dx < -40 && ci < events.length - 1) ci++;
    else if (dx > 40 && ci > 0) ci--;
    updC();
  };
}

function chgCnt(d) {
  var oldCnt = cnt;
  cnt = Math.max(1, Math.min(20, cnt + d));
  document.getElementById('cntVal').textContent = cnt;
  // Animasyonlu: arttırınca yeni alan açılır, azaltınca kapanır
  if (cnt > oldCnt) {
    renderGuestFields(cnt - 1); // sadece yeni alanı ekle
  } else if (cnt < oldCnt) {
    removeLastGuest(oldCnt - 1);
  }
}

function renderGuestFields(targetIndex) {
  // targetIndex: eklenecek misafirin numarası (1'den başlar)
  var wrap = document.getElementById('guestNamesWrap');
  if (!wrap) return;

  // Eğer targetIndex verilmemişse tümünü yeniden render et
  if (targetIndex === undefined) {
    wrap.innerHTML = '';
    for (var i = 1; i < cnt; i++) {
      addGuestField(wrap, i, false);
    }
    return;
  }

  addGuestField(wrap, targetIndex, true);
}

function addGuestField(wrap, index, animate) {
  var div = document.createElement('div');
  div.className = 'f';
  div.id = 'guestField' + index;
  if (animate) {
    div.style.cssText = 'overflow:hidden;max-height:0;opacity:0;transition:max-height .3s ease, opacity .25s ease;margin-bottom:0';
  }
  var lbl = document.createElement('label');
  lbl.textContent = index + '. MİSAFİR';
  var inp = document.createElement('input');
  inp.className = 'fi';
  inp.type = 'text';
  inp.id = 'fGuest' + index;
  inp.placeholder = index + '. misafirin adı soyadı';
  div.appendChild(lbl);
  div.appendChild(inp);
  wrap.appendChild(div);

  if (animate) {
    // Bir sonraki frame'de animasyonu başlat
    requestAnimationFrame(function() {
      requestAnimationFrame(function() {
        div.style.maxHeight = '80px';
        div.style.opacity = '1';
        div.style.marginBottom = '';
        // Animasyon bitince inline style temizle
        setTimeout(function() {
          div.style.cssText = '';
          div.className = 'f';
        }, 350);
        // Yeni alana fokus ver
        inp.focus();
      });
    });
  }
}

function removeLastGuest(index) {
  var field = document.getElementById('guestField' + index);
  if (!field) {
    // ID'li bulamazsa wrap'ın son çocuğunu kaldır
    var wrap = document.getElementById('guestNamesWrap');
    if (wrap && wrap.lastChild) {
      var last = wrap.lastChild;
      last.style.cssText = 'overflow:hidden;max-height:80px;opacity:1;transition:max-height .25s ease, opacity .2s ease';
      requestAnimationFrame(function() {
        last.style.maxHeight = '0';
        last.style.opacity = '0';
        setTimeout(function() { if (last.parentNode) last.parentNode.removeChild(last); }, 260);
      });
    }
    return;
  }
  field.style.cssText = 'overflow:hidden;max-height:80px;opacity:1;transition:max-height .25s ease, opacity .2s ease';
  requestAnimationFrame(function() {
    field.style.maxHeight = '0';
    field.style.opacity = '0';
    setTimeout(function() { if (field.parentNode) field.parentNode.removeChild(field); }, 260);
  });
}

async function doRes() {
  if (!me) { showAuthModal(); return; }
  var name = document.getElementById('fName').value.trim();
  var phone = document.getElementById('fPhone').value.trim();
  if (!selEvt) { toast('Lütfen bir etkinlik seçin'); return; }
  if (!name) { toast('Üye bilgisi yüklenemedi, lütfen tekrar giriş yapın'); return; }
  var rw = document.getElementById('rulesWrap');
  var rc = document.getElementById('rulesCheck');
  if (rw && rw.style.display !== 'none' && rc && !rc.checked) {
    rw.style.borderLeftColor = 'var(--red)';
    setTimeout(function() { rw.style.borderLeftColor = ''; }, 2000);
    toast('Lütfen etkinlik kurallarını onaylayın');
    return;
  }
  var e = events.find(function(x) { return x.id === selEvt; });
  var qrId = 'NIP-' + Math.random().toString(36).substr(2, 6).toUpperCase();
  var btn = document.getElementById('subBtn');
  btn.disabled = true;
  btn.innerHTML = '<span id="subBtnTxt">GÖNDERİLİYOR...</span>';
  var guestNames = [];
  for (var gi = 1; gi < cnt; gi++) {
    var gInp = document.getElementById('fGuest' + gi);
    if (gInp) {
      var gVal = gInp.value.trim();
      if (!gVal) {
        toast(gi + '. misafirin adını girin');
        // O alana scroll yap
        gInp.scrollIntoView({ behavior: 'smooth', block: 'center' });
        gInp.focus();
        btn.disabled = false;
        btn.innerHTML = '<span id="subBtnTxt">' + t('resBtn') + '</span>';
        return;
      }
      guestNames.push(gVal);
    }
  }
  var ins = await _SB.from('reservations').insert({
    qr_id: qrId, event_id: e.id, event_name: e.name,
    event_date: e.date, event_time: e.time, name: name, phone: phone,
    count: cnt, status: 'pending', member_id: me.id,
    member_code: me.member_code, rules_accepted: rc ? rc.checked : false,
    guest_name: guestNames.length > 0 ? guestNames.join(', ') : null
  }).select().single();
  btn.disabled = false;
  btn.innerHTML = '<span id="subBtnTxt">' + t('resBtn') + '</span>';
  if (ins.error) { toast('Hata: ' + ins.error.message); return; }
  G('s-tickets');
  openTickets();
}

// ── YENİ: Arama + filtreleme destekli renderM ──
async function renderM() {
  var search = (document.getElementById('mSearch') ? document.getElementById('mSearch').value : '').trim().toLowerCase();
  var sortVal = document.getElementById('mSort') ? document.getElementById('mSort').value : 'created_at_desc';
  var refFilter = window._mRefFilter || '';

  var sortField = sortVal.startsWith('name') ? 'name' : 'created_at';
  var sortAsc = sortVal.endsWith('_asc');

  var q = _SB.from('members').select('*').order(sortField, { ascending: sortAsc });
  if (mF) q = q.eq('status', mF);

  var res = await q;
  var data = res.data || [];

  // Referans chip'lerini güncelle (tüm veriyle)
  renderRefChips(data);

  // Client-side arama: isim, email, telefon, Instagram
  if (search) {
    data = data.filter(function(m) {
      return (m.name || '').toLowerCase().includes(search) ||
             (m.email || '').toLowerCase().includes(search) ||
             (m.phone || '').toLowerCase().includes(search) ||
             (m.instagram || '').toLowerCase().includes(search);
    });
  }

  // Referans filtresi
  if (refFilter) {
    data = data.filter(function(m) { return (m.ref_by || '') === refFilter; });
  }

  var list = document.getElementById('mList');
  if (!data.length) {
    list.innerHTML = '<div class="empty">' +
      (search ? '"' + search + '" için sonuç bulunamadı.' : 'Kayıt yok.') + '</div>';
    return;
  }

  renderMCards(data, 'mList');

  // Arama aktifse sonuç sayısını göster
  if (search || refFilter) {
    var info = document.createElement('div');
    info.style.cssText = 'font-family:Space Mono,monospace;font-size:7px;color:#555;text-align:right;padding:4px 0 8px;letter-spacing:1px';
    info.textContent = data.length + ' SONUÇ';
    list.insertBefore(info, list.firstChild);
  }
}

// YENİ: Referans chip'lerini render et
function renderRefChips(allMembers) {
  var wrap = document.getElementById('mRefChips');
  if (!wrap) return;

  // Benzersiz referans veren kişiler
  var refs = {};
  allMembers.forEach(function(m) { if (m.ref_by) refs[m.ref_by] = (refs[m.ref_by] || 0) + 1; });
  var refNames = Object.keys(refs);

  if (!refNames.length) { wrap.innerHTML = ''; return; }

  var current = window._mRefFilter || '';
  var html = '<button class="chip' + (!current ? ' active' : '') +
    '" onclick="setRefFilter(this,\'\')" style="font-size:6px">TÜMÜ</button>';

  refNames.sort().forEach(function(name) {
    var count = refs[name];
    var safe = name.replace(/\\/g, '\\\\').replace(/'/g, "\\'");
    html += '<button class="chip' + (current === name ? ' active' : '') +
      '" onclick="setRefFilter(this,\'' + safe + '\')" style="font-size:6px">' +
      name.toUpperCase() + ' <span style="opacity:.5">(' + count + ')</span></button>';
  });

  wrap.innerHTML = html;
}

async function compressAllPosters() {
  var btn = document.getElementById('compressAllBtn');
  if (btn) { btn.textContent = '⏳ KONTROL EDİLİYOR...'; btn.disabled = true; }

  var res = await _SB.from('events').select('id, name, poster_b64');
  var events_list = (res.data || []).filter(function(e) { return e.poster_b64; });

  if (!events_list.length) {
    toast('Afiş bulunan etkinlik yok');
    if (btn) { btn.textContent = '⚡ MEVCUT AFİŞLERİ SIKIŞIR'; btn.disabled = false; }
    return;
  }

  var totalOrig = 0, totalComp = 0, skipped = 0, updated = 0;

  for (var i = 0; i < events_list.length; i++) {
    var evt = events_list[i];
    var origKB = b64Size(evt.poster_b64);
    totalOrig += origKB;

    if (btn) btn.textContent = '⏳ ' + (i+1) + '/' + events_list.length + ' ' + evt.name.slice(0,12) + '...';

    // Zaten küçükse atla (150KB altı JPEG muhtemelen zaten sıkıştırılmış)
    if (origKB < 150 && evt.poster_b64.startsWith('data:image/jpeg')) {
      totalComp += origKB;
      skipped++;
      continue;
    }

    var compressed = await compressImage(evt.poster_b64, 900, 0.78);
    var compKB = b64Size(compressed);
    totalComp += compKB;

    // Sadece gerçekten küçüldüyse kaydet (en az %10 küçülme)
    if (compKB < origKB * 0.9) {
      await _SB.from('events').update({ poster_b64: compressed }).eq('id', evt.id);
      updated++;
    } else {
      skipped++;
    }
  }

  var saved = totalOrig - totalComp;
  var savedMB = saved > 1024 ? (saved/1024).toFixed(1) + 'MB' : saved + 'KB';
  toast('✓ ' + updated + ' afiş güncellendi · ' + savedMB + ' kazanıldı · ' + skipped + ' atlandı');

  if (btn) { btn.textContent = '✓ TAMAMLANDI'; btn.disabled = false; }
  setTimeout(function() {
    if (btn) btn.textContent = '⚡ MEVCUT AFİŞLERİ SIKIŞIR';
  }, 3000);

  if (updated > 0) renderEvts();
}

// ── GÖRSEL SIKIŞTIRIC ──
// maxW: max genişlik (px), quality: 0-1 JPEG kalitesi
// Döndürür: Promise<base64string>
function compressImage(src, maxW, quality) {
  return new Promise(function(resolve) {
    var img = new Image();
    img.onload = function() {
      var w = img.width, h = img.height;
      // 4:5 oranına crop etmeden, sadece max genişliğe küçült
      if (w > maxW) {
        h = Math.round(h * maxW / w);
        w = maxW;
      }
      var canvas = document.createElement('canvas');
      canvas.width = w;
      canvas.height = h;
      var ctx = canvas.getContext('2d');
      ctx.drawImage(img, 0, 0, w, h);
      resolve(canvas.toDataURL('image/jpeg', quality));
    };
    img.onerror = function() { resolve(src); }; // Hata olursa orijinali döndür
    img.src = src;
  });
}

// Boyutu KB cinsinden hesapla
function b64Size(b64) {
  return Math.round((b64.length * 3) / 4 / 1024);
}

function onPoster(ev) {
  var f = ev.target.files[0]; if (!f) return;
  var uzBtn = document.getElementById('uZone');
  if (uzBtn) uzBtn.style.opacity = '.4';
  var r = new FileReader();
  r.onload = async function(e2) {
    var original = e2.target.result;
    var origKB = b64Size(original);
    // Max 900px genişlik, %78 JPEG kalitesi
    var compressed = await compressImage(original, 900, 0.78);
    var compKB = b64Size(compressed);
    newP = compressed;
    document.getElementById('pPrev').style.display = 'block';
    document.getElementById('pPrevImg').src = newP;
    if (uzBtn) { uzBtn.style.display = 'none'; uzBtn.style.opacity = '1'; }
    toast('Afiş yüklendi · ' + origKB + 'KB → ' + compKB + 'KB (' + Math.round(compKB/origKB*100) + '%) ✓');
  };
  r.readAsDataURL(f);
}

      toast('Afiş güncellendi · ' + origKB + 'KB → ' + compKB + 'KB ✓');
      renderEvts();
    };
    r.readAsDataURL(f);
  };
  inp.click();
}

async function startCam() {
  try {
    camStream = await navigator.mediaDevices.getUserMedia({video:{facingMode:'environment',width:{ideal:640},height:{ideal:640}}});
    var v = document.getElementById('qrVideo'); v.srcObject = camStream; v.style.display = 'block'; v.play();
    document.getElementById('camOff').style.display = 'none';
    document.getElementById('camOn').style.display = 'block';
    document.getElementById('qrRes').style.display = 'none';
    camTimer = setInterval(scanFrame, 300);
  } catch(e) { toast('Kamera erişimi reddedildi'); }
}
function stopCam() {
  if (camStream) { camStream.getTracks().forEach(function(t) { t.stop(); }); camStream = null; }
  if (camTimer) { clearInterval(camTimer); camTimer = null; }
  var v = document.getElementById('qrVideo'); if (v) { v.style.display = 'none'; v.srcObject = null; }
  var co = document.getElementById('camOff'); if (co) co.style.display = 'block';
  var cn = document.getElementById('camOn'); if (cn) cn.style.display = 'none';
}
function scanFrame() {
  var v = document.getElementById('qrVideo');
  if (!v || v.readyState !== v.HAVE_ENOUGH_DATA) return;
  var c = document.getElementById('qrCanvas'); c.width = v.videoWidth; c.height = v.videoHeight;
  var ctx = c.getContext('2d'); ctx.drawImage(v, 0, 0, c.width, c.height);
  var img = ctx.getImageData(0, 0, c.width, c.height);
  var code = jsQR(img.data, img.width, img.height, {inversionAttempts:'dontInvert'});
  if (code && code.data && code.data.startsWith('NIP-')) { stopCam(); document.getElementById('qrInput').value = code.data; scanQR(); }
}

function fmtDate(s) {
  if (!s) return '';
  try { var d = new Date(s + 'T12:00:00'); return d.toLocaleDateString('tr-TR', {day:'numeric',month:'short',year:'numeric'}); } catch(e) { return s; }
}

(function(){
  var startY = 0, startX = 0, tracking = false, startedOnSheet = false;
  
  document.addEventListener('touchstart', function(e){
    var touch = e.touches[0];
    startY = touch.clientY;
    startX = touch.clientX;
    tracking = true;
    
    // Dokunuşun sheet üzerinde mi yoksa ekranın alt bölgesinde mi başladığını kontrol et
    var sheet = document.getElementById('resSheet');
    var winH = window.innerHeight;
    var isBottomZone = touch.clientY > winH * 0.65; // Ekranın alt %35'i
    startedOnSheet = sheet && (sheet.contains(e.target) || isBottomZone);
  }, {passive:true});

  document.addEventListener('touchmove', function(e){
    if (!tracking) return;
    var sheet = document.getElementById('resSheet');
    if (!sheet) return;
    var dy = e.touches[0].clientY - startY;
    var dx = Math.abs(e.touches[0].clientX - startX);
    if (dx > Math.abs(dy) && dx > 15) { tracking = false; return; }
    // Sheet açıkken aşağı kaydırma ve scroll en üsteyse sayfayı engelle
    if (sheet.classList.contains('open') && dy > 0 && sheet.scrollTop === 0) {
      e.preventDefault();
    }
  }, {passive:false});

  document.addEventListener('touchend', function(e){
    if (!tracking) return;
    tracking = false;
    var sheet = document.getElementById('resSheet');
    if (!sheet) return;
    var dy = e.changedTouches[0].clientY - startY;
    var dx = Math.abs(e.changedTouches[0].clientX - startX);
    // Yatay swipe ise yoksay
    if (dx > Math.abs(dy) || dx > 40) return;
    
    if (!sheet.classList.contains('open')) {
      // Kapalıyken: alt bölgeden VEYA sheet handle üzerinden yukarı swipe → aç
      if (dy < -30 && startedOnSheet) {
        openSheet();
      }
    } else {
      // Açıkken: aşağı swipe + scroll en üstte → kapat
      if (dy > 60 && sheet.scrollTop === 0) {
        closeSheet();
      }
    }
  }, {passive:true});
})();

boot();
</script>
